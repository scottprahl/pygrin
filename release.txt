Releasing a new version

# make sure things are not broken and that files match up
  make rcheck

# update the version in __init__.py and pyproject.toml
  git commit pygrin/__init__.py pyproject.toml -m 'update version'

# update CHANGELOG.txt  `git shortlog 0.6.1..HEAD`
  git commit -m 'update recent changes' CHANGELOG.txt
  git push

# deploy JupyterLite files
  make realclean
  make lite
  make lite-deploy

# create release tag on github
  git tag 0.6.0
  git push origin 0.6.0

# manually publish release at
  open https://github.com/scottprahl/pygrin
  
# verify action success
  open https://pypi.org/project/pygrin/
  open https://pygrin.readthedocs.io
  
# wait for feedstock update
  open https://github.com/conda-forge/pygrin-feedstock
