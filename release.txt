Releasing a new version

# versioning
update the version in setup.py and pygrin/__init__.py

  git commit setup.py pygrin/__init__.py -m 'update version'
  git push

update CHANGELOG.txt using `git shortlog`
  git commit CHANGELOG.txt -m 'update recent changes' 
  git push

# create release tag on github
  git tag v1.0.1
  git push origin v1.0.1

# upload source to pypi
  rm -rf dist/*
  python setup.py sdist
  twine upload dist/*

# test the install by removing local developer version
# installing, then uninstalling, then restoring developer version
  conda develop -u .
  pip install pygrin
  pip uninstall pygrin
  conda develop .
